
\renewcommand{\arraystretch}{1.5}

\begin{longtable}{l p{13.7cm}}
\textbf{API} & \textit{Application Programming Interface}. Interfaz de programación de aplicaciones. \\
\textbf{CDN} & \textit{Content Delivery Network}. Red de distribución de contenidos. \\
\textbf{CLS} & \textit{Cumulative Layout Shift}. Métrica de estabilidad visual de Core Web Vitals. \\
\textbf{CSRF} & \textit{Cross-Site Request Forgery}. Tipo de ataque web que fuerza acciones no deseadas. \\
\textbf{FID} & \textit{First Input Delay}. Tiempo de respuesta al primer input del usuario. \\
\textbf{GraphQL} & Lenguaje de consulta para APIs desarrollado por Meta. \\
\textbf{httpOnly} & Atributo de cookie que impide su acceso desde JavaScript del navegador. \\
\textbf{ISR} & \textit{Incremental Static Regeneration}. Estrategia de caché de Next.js. \\
\textbf{JWT} & \textit{JSON Web Token}. Estándar para la transmisión segura de información entre partes. \\
\textbf{LCP} & \textit{Largest Contentful Paint}. Métrica de velocidad de carga de Core Web Vitals. \\
\textbf{OAuth} & \textit{Open Authorization}. Protocolo estándar de delegación de acceso. \\
\textbf{OMDb} & \textit{Open Movie Database}. Base de datos de películas con datos de IMDb. \\
\textbf{REST} & \textit{Representational State Transfer}. Estilo arquitectónico para servicios web. \\
\textbf{RSC} & \textit{React Server Components}. Componentes React que se ejecutan exclusivamente en servidor. \\
\textbf{SPA} & \textit{Single Page Application}. Aplicación web de una sola página. \\
\textbf{SPR} & \textit{Serverless Page Rendering}. Renderizado de páginas sin servidor propio. \\
\textbf{SSR} & \textit{Server-Side Rendering}. Renderizado de páginas en el servidor. \\
\textbf{TMDb} & \textit{The Movie Database}. Base de datos de películas y series con API pública. \\
\textbf{TTL} & \textit{Time To Live}. Tiempo de vida de un elemento en caché. \\
\textbf{XSS} & \textit{Cross-Site Scripting}. Tipo de ataque de inyección de código en el navegador. \\
\end{longtable}
